00619 BF_FocalElement BFS_getBeliefValue(const BFS_PartOfBelief pob, const double sensorMeasure, const int elementSize){
00620     BF_FocalElement point = {{NULL, 0}, 0};
00621     int i = 0;
00622     float approximation = 0;
00623 
00624     /*Save the focal element: */
00625     point.element = Sets_copyElement(pob.focalElement, elementSize);
00626     /*If the first point is bigger: */
00627     if(sensorMeasure <= pob.points[0].sensorValue){
00628         point.beliefValue = pob.points[0].belief;
00629     }
00630     /*If the last point is smaller: */
00631     else if (sensorMeasure >= pob.points[pob.nbPts - 1].sensorValue){
00632         point.beliefValue = pob.points[pob.nbPts - 1].belief;
00633     }
00634     else {
00635         /*Linear approximation: */
00636         for(i = 0; i < pob.nbPts - 1; i++){
00637             if(pob.points[i].sensorValue <= sensorMeasure && sensorMeasure <= pob.points[i+1].sensorValue){
00638                 approximation = ((pob.points[i+1].belief - pob.points[i].belief) *
00639                                  (sensorMeasure - pob.points[i].sensorValue) /
00640                                  (pob.points[i+1].sensorValue - pob.points[i].sensorValue));
00641                 point.beliefValue = pob.points[i].belief + approximation;
00642                 break;
00643             }
00644         }
00645     }
00646 
00647     return point;
